#!/bin/bash
LOGFILE="/var/log/wifi-status.log"

{
  echo "==== $(date) ===="
  iwgetid
  ip -brief addr show wlan0
  ip route
  echo
} >> "$LOGFILE" 2>&1

sudo chmod +x /usr/local/bin/log-wifi-status.sh

[Unit]
Description=Log Wi-Fi SSID and IP at boot
After=network-online.target

[Service]
Type=oneshot
ExecStart=/usr/local/bin/log-wifi-status.sh

[Install]
WantedBy=multi-user.target


sudo systemctl enable log-wifi-status.service
